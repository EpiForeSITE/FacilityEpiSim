001//this is a data class to hold infomration about discharged patients.
002//we need all relevant information from the discharged person class, as well as 
003//the information from the person's PersonDisease object.  Assume only one disease per person for now.
004
005package agents;
006
007public class DischargedPatient {
008        
009        // Patient identification and timing information
010        private int id;
011        private double admissionTime;
012        private double dischargeTime;
013        private double los;
014        
015        // Patient status information
016        private boolean isIsolated;
017        private boolean isColonized;
018        private boolean isDetected;
019        private boolean detectedBySurveillance;
020        private boolean clinicallyDetectedDuringCurrentStay;
021        private boolean initialInfection;
022        private int detectionCount;
023        
024        // Disease-related information
025        private String diseaseName;
026        private double transmissionRateContribution;
027        
028        // Constructors
029        public DischargedPatient() {
030                // Default constructor
031        }
032        
033        public DischargedPatient(Person person) {
034                this.id = person.hashCode(); // Using hashCode as ID
035                this.admissionTime = person.getAdmissionTime();
036                this.dischargeTime = person.getDischargeTime();
037                this.los = this.dischargeTime - this.admissionTime;
038                this.isIsolated = person.isIsolated();
039                
040                // Extract disease information (assuming one disease per person)
041                if (!person.getDiseases().isEmpty()) {
042                        var personDisease = person.getDiseases().get(0);
043                        this.isColonized = personDisease.isColonized();
044                        this.isDetected = personDisease.isDetected();
045                        this.detectedBySurveillance = personDisease.isDetectedBySurveillance();
046                        this.clinicallyDetectedDuringCurrentStay = personDisease.isClinicallyDetectedDuringCurrentStay();
047                        this.initialInfection = personDisease.isInitialInfection();
048                        this.detectionCount = personDisease.getDetectionCount();
049                        this.diseaseName = personDisease.getDisease().getDiseaseName();
050                        this.transmissionRateContribution = personDisease.getTransmissionRateContribution();
051                }
052        }
053        
054        // Getters and Setters
055        public int getId() {
056                return id;
057        }
058        
059        public void setId(int id) {
060                this.id = id;
061        }
062        
063        public double getAdmissionTime() {
064                return admissionTime;
065        }
066        
067        public void setAdmissionTime(double admissionTime) {
068                this.admissionTime = admissionTime;
069        }
070        
071        public double getDischargeTime() {
072                return dischargeTime;
073        }
074        
075        public void setDischargeTime(double dischargeTime) {
076                this.dischargeTime = dischargeTime;
077        }
078        
079        public double getLos() {
080                return los;
081        }
082        
083        public void setLos(double los) {
084                this.los = los;
085        }
086        
087        public boolean isIsolated() {
088                return isIsolated;
089        }
090        
091        public void setIsolated(boolean isIsolated) {
092                this.isIsolated = isIsolated;
093        }
094        
095        public boolean isColonized() {
096                return isColonized;
097        }
098        
099        public void setColonized(boolean isColonized) {
100                this.isColonized = isColonized;
101        }
102        
103        public boolean isDetected() {
104                return isDetected;
105        }
106        
107        public void setDetected(boolean isDetected) {
108                this.isDetected = isDetected;
109        }
110        
111        public boolean isDetectedBySurveillance() {
112                return detectedBySurveillance;
113        }
114        
115        public void setDetectedBySurveillance(boolean detectedBySurveillance) {
116                this.detectedBySurveillance = detectedBySurveillance;
117        }
118        
119        public boolean isClinicallyDetectedDuringCurrentStay() {
120                return clinicallyDetectedDuringCurrentStay;
121        }
122        
123        public void setClinicallyDetectedDuringCurrentStay(boolean clinicallyDetectedDuringCurrentStay) {
124                this.clinicallyDetectedDuringCurrentStay = clinicallyDetectedDuringCurrentStay;
125        }
126        
127        public boolean isInitialInfection() {
128                return initialInfection;
129        }
130        
131        public void setInitialInfection(boolean initialInfection) {
132                this.initialInfection = initialInfection;
133        }
134        
135        public int getDetectionCount() {
136                return detectionCount;
137        }
138        
139        public void setDetectionCount(int detectionCount) {
140                this.detectionCount = detectionCount;
141        }
142        
143        public String getDiseaseName() {
144                return diseaseName;
145        }
146        
147        public void setDiseaseName(String diseaseName) {
148                this.diseaseName = diseaseName;
149        }
150        
151        public double getTransmissionRateContribution() {
152                return transmissionRateContribution;
153        }
154        
155        public void setTransmissionRateContribution(double transmissionRateContribution) {
156                this.transmissionRateContribution = transmissionRateContribution;
157        }
158        
159        @Override
160        public String toString() {
161                return String.format("%d,%.2f,%.2f,%.2f,%b,%b,%b,%b,%b,%b,%d,%s,%.6f", 
162                                id, admissionTime, dischargeTime, los, isIsolated, isColonized, isDetected, 
163                                detectedBySurveillance, clinicallyDetectedDuringCurrentStay, initialInfection, 
164                                detectionCount, diseaseName, transmissionRateContribution);
165        }
166        
167        /**
168         * Returns a CSV header string containing the names of all member variables.
169         * This can be used as the first line in a CSV file containing discharged patient data.
170         * 
171         * @return CSV header string with comma-separated field names
172         */
173        public static String getHeader() {
174                return "id,admissionTime,dischargeTime,los,isIsolated,isColonized,isDetected," +
175                                "detectedBySurveillance,clinicallyDetectedDuringCurrentStay,initialInfection," +
176                                "detectionCount,diseaseName,transmissionRateContribution";
177        }
178}

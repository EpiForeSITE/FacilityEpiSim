001package utils;
002
003import repast.simphony.random.RandomHelper;
004import java.util.List;
005
006/**
007 * Utility class for random selection from lists based on probabilities.
008 * <p>
009 * Provides static methods for randomly choosing from a set of options,
010 * either with uniform probability or weighted by density values.
011 * </p>
012 *
013 * @author [Project Team]
014 * @version 1.0
015 */
016public class Chooser {
017
018        /**
019         * Randomly selects an option from a list based on weighted probabilities (density).
020         * <p>
021         * The density list should contain probability values that sum to 1.0.
022         * Each density value corresponds to the probability of selecting the option
023         * at that index in the options list.
024         * </p>
025         *
026         * @param options the list of options to choose from
027         * @param density the probability density for each option
028         * @return the randomly selected option
029         */
030        public static Object choose(List<?> options, List<? extends Number> density) {
031                double testValue = RandomHelper.getUniform().nextDoubleFromTo(0.0, 1.0);
032
033                double cumulative = 0;
034                int step = 0;
035                int choiceStep = -1;
036
037                for (Number d : density) {
038                        cumulative += d.doubleValue();
039                        if (testValue <= cumulative) {
040                                choiceStep = step;
041                                break;
042                        }
043                        step++;
044                }
045
046                return options.get(choiceStep);
047        }
048
049        /**
050         * Randomly selects one option from a list with uniform probability.
051         *
052         * @param options the list of options to choose from
053         * @return a randomly selected option
054         */
055        public static Object chooseOne(List<?> options) {
056                int index = RandomHelper.getUniform().nextIntFromTo(0, options.size() - 1);
057                return options.get(index);
058        }
059
060        /**
061         * Randomly returns true or false with equal probability (50/50).
062         *
063         * @return a random boolean value
064         */
065        public static boolean coinFlip() {
066                List<Boolean> options = java.util.Arrays.asList(true, false);
067                Object choice = chooseOne(options);
068                return (Boolean) choice;
069        }
070
071        /**
072         * Returns true with probability {@code test}, false with probability {@code 1.0 - test}.
073         *
074         * @param test the probability of returning true (must be in range [0, 1])
075         * @return true with probability test, false otherwise
076         * @throws IllegalArgumentException if test is outside the range [0, 1]
077         */
078        public static boolean randomTrue(Double test) {
079                if (test < 0 || test > 1) {
080                        throw new IllegalArgumentException("value " + test + " out of bounds [0,1]");
081                }
082                List<Boolean> options = java.util.Arrays.asList(true, false);
083                List<Double> density = java.util.Arrays.asList(test, 1.0 - test);
084                Object choice = choose(options, density);
085                return (Boolean) choice;
086        }
087
088}

023/**
024 * Represents a healthcare facility agent in the disease transmission simulation.
025 * <p>
026 * {@code Facility} is the primary container for managing patient agents, disease outbreaks,
027 * and healthcare operations. It tracks patient admissions, discharges, disease transmission,
028 * and surveillance testing within a single facility. The facility operates within a {@link Region}
029 * that coordinates multi-facility interactions if needed.
030 * Key responsibilities include:
031 * </p>
032 * <ul>
033 *   <li>Managing patient admission and discharge lifecycle</li>
034 *   <li>Maintaining disease outbreak tracking via {@link FacilityOutbreak} agents</li>
035 *   <li>Updating transmission rates based on current patient population</li>
036 *   <li>Recording surveillance testing and clinical detection events</li>
037 *   <li>Tracking statistics (patient days, admissions, population averages)</li>
038 * </ul>
039 * <p>
040 * The facility is configured with parameters including isolation effectiveness, surveillance
041 * adherence rates, and mean length of stay distributions specific to facility type.
042 * </p>
043 *
044 * @author [Project Team]
045 * @version 1.0
046 * @see Region
047 * @see FacilityOutbreak
048 * @see Person
049 * @see Disease
050 */
051public class Facility extends AgentContainer{

121        /**
122         * Admits an existing patient to this facility.
123         * Integrates a patient into facility operations by:
124         * <ul>
125         *   <li>Adding the patient to the facility's region and patient list</li>
126         *   <li>Starting a discharge timer based on random length of stay</li>
127         *   <li>Testing for colonized diseases via surveillance (if active surveillance enabled)</li>
128         *   <li>Starting clinical detection timers for any colonized diseases</li>
129         *   <li>Scheduling periodic surveillance tests (if enabled)</li>
130         *   <li>Updating transmission rate contributions</li>
131         *   <li>Recording admission statistics (if in measurement period)</li>
132         * </ul>
133         *
134         * @param p the patient to admit
135         */
136        public void admitPatient(Person p){
137                region.importToFacilityNew(this,p);
138            
139                logPatientAdmission(schedule.getTickCount(), p.hashCode(), (boolean) p.getProperty("importation"));
140                p.admitToFacility(this);
141                
142
143                p.startDischargeTimer(getRandomLOS());
144
145                
146
147                for(PersonDisease pd : p.getDiseases()){
148                        if(pd.isColonized()){
149                                
150                                if(pd.getDisease().isActiveSurveillanceAgent() && onActiveSurveillance){
151                                        if(uniform() < pd.getDisease().getProbSurveillanceDetection() * admissionSurveillanceAdherence){
152                                                pd.setDetected(true);
153                                                pd.setDetectedBySurveillance();
154                                                if(pd.getDisease().isolatePatientWhenDetected()) p.isolate();
155                                        }
156                                }
157                                pd.startClinicalDetectionTimer();
158                        }
159                }
160                getCurrentPatients().add(p);
161                getRegion().getPeople().add(p);
162
163                if(onActiveSurveillance && !p.isIsolated() && getTimeBetweenMidstaySurveillanceTests() > 0)
164                        
165                        p.doSurveillanceTest();
166                        p.startNextPeriodicSurveillanceTimer();
167
168                p.updateAllTransmissionRateContributions();
169
170                if(!getRegion().isInBurnInPeriod()) updateAdmissionTally(p);
171        }
172        /**
173         * Discharges a patient from this facility.
174         * Removes a patient from active operations by:
175         * <ul>
176         *   <li>Removing the patient from the facility's region and patient list</li>
177         *   <li>Recording the discharge time</li>
178         *   <li>Creating a {@link agents.DischargedPatient} record (if in measurement period)</li>
179         *   <li>Updating stay statistics for tracking patient-days (if in measurement period)</li>
180         *   <li>Canceling all scheduled events for the patient</li>
181         *   <li>Removing the patient from the Repast context for garbage collection</li>
182         *   <li>Updating transmission rates across all disease outbreaks</li>
183         * </ul>
184         *
185         * @param p the patient to discharge
186         */
187        public void dischargePatient(Person p){
188                region.people.remove(p);
189
190
191                getCurrentPatients().remove(p);
192                updateTransmissionRate();
193                SingleFacilityBuilder builder = getSimulationBuilder();
194                p.setDischargeTime(TimeUtils.getSchedule().getTickCount());
195                if (!region.isInBurnInPeriod()) {
196                        builder.dischargedPatients.add(new agents.DischargedPatient(p));
197                }
198                if(!getRegion().isInBurnInPeriod()) updateStayTally(p);
199                p.destroyMyself(getRegion());
200                // Remove from Repast context to allow dereferencing and garbage collection
201                if (builder.getContext() != null) {
202                        builder.getContext().remove(p);
203                }
204                p.setNoMoreEvents(true);
205        }

219        /**
220         * Generates a random length of stay (LOS) for a patient based on facility type.
221         * <p>
222         * For type 0 facilities (long-term acute care), uses a mixture of two gamma distributions
223         * to model realistic LOS patterns. Other facility types return -1.0.
224         * </p>
225         *
226         * @return the random length of stay in days, or -1.0 if facility type is not recognized
227         */
228        public double getRandomLOS(){
229                if(getType()==0){
230
231                        double shape1 = 7.6019666;
232                        double scale1 = 3.4195217;
233                        double shape2 = 1.2327910;
234                        double scale2 = 23.5214724;
235                        double prob1 = 0.6253084;
236
237                        if(uniform() < prob1) return gamma(shape1,scale1);
238                        else return gamma(shape2,scale2);
239                }
240                else{
241                        return -1.0;
242                }
243        }

245        /**
246         * Admits an initial patient during the facility setup phase (burn-in period).
247         * <p>
248         * Similar to {@link #admitPatient(Person)} but without surveillance testing or
249         * admission statistics tracking, as these patients are used to establish a stable
250         * population during the burn-in period. The patient's discharge is scheduled based
251         * on the facility's mean length of stay.
252         * </p>
253         *
254         * @param p the initial patient to admit
255         */
256        public void admitInitialPatient(Person p){
257                p.admitToFacility(this);
258                p.startDischargeTimer(exponential(1.0/getMeanLOS()));
259
260                region.people.add(p);
261
262                if(onActiveSurveillance) {
263                }
264
265                for(PersonDisease pd : p.getDiseases()){
266                        if(pd.isColonized()){
267                                pd.startClinicalDetectionTimer();
268                        }
269                }
270                getCurrentPatients().add(p);
271
272                p.updateAllTransmissionRateContributions();
273        }

275        /**
276         * Updates the daily population statistics for this facility.
277         * <p>
278         * Calculates the running average population size and updates prevalence tallies
279         * for each disease outbreak. Called once per simulation day.
280         * </p>
281         */
282        public void updatePopulationTally(){
283                avgPopulation = (avgPopulation * numDaysTallied + region.people.size() / (numDaysTallied + 1));
284                numDaysTallied++;
285
286                for(FacilityOutbreak fo : outbreaks) {
287                        fo.updatePrevalenceTally();
288                }
289
290        }

292        /**
293         * Updates statistics tracking patient-days for a discharged patient.
294         * <p>
295         * Increments the total patient-days counter and updates outbreak-specific
296         * stay statistics for each disease tracked by the facility.
297         * </p>
298         *
299         * @param p the discharged patient
300         */
301        public void updateStayTally(Person p){
302                setPatientDays(getPatientDays() + p.getCurrentLOS());
303
304
305                if(!outbreaks.isEmpty()&&!p.personDiseases.isEmpty()) {
306                for(int i=0; i<outbreaks.size(); i++) {
307                        outbreaks.get(i).updateStayTally(p.personDiseases.get(i));
308                        }
309                }
310        }

312        /**
313         * Updates statistics tracking admissions and disease states.
314         * <p>
315         * Increments the admission counter and updates outbreak-specific admission tallies
316         * for each disease tracked by the facility. Admission statistics are used to
317         * calculate disease importation rates and endemic transmission patterns.
318         * </p>
319         *
320         * @param p the admitted patient
321         */
322        public void updateAdmissionTally(Person p){
323            // Jan 10, 2025 WRR: it's appropriate to count the number of admissions in an int...
324            // because it's a total count of all the admissions EVER, and most of them have been
325            // discharged.  Things like current population size, or percentage of patients colonized
326            // should wherever possible be calculated from the relevant collection of "live" patients
327                numAdmissions++;
328
329                if(!outbreaks.isEmpty()&&!p.personDiseases.isEmpty()) {
330                        for(int i=0; i<outbreaks.size(); i++) {
331                                outbreaks.get(i).updateAdmissionTally(p.personDiseases.get(i));
332                        }
333                }
334        }

513    /**
514     * Gets a reference to the root context in Repast Simphony.
515     * This uses ContextUtils to get the context containing this facility object,
516     * which should be the root context where the simulation builder was added.
517     * 
518     * @return the root context
519     */
520    @SuppressWarnings("unchecked")
521    public Context<Object> getRootContext() {
522        return ContextUtils.getContext(this);
523    }
524    
525    /**
526     * Gets the SingleFacilityBuilder from the root context.
527     * This allows access to the main simulation controller and its methods/data.
528     * 
529     * @return the SingleFacilityBuilder instance, or null if not found
530     */
531    public SingleFacilityBuilder getSimulationBuilder() {
532        Context<Object> rootContext = getRootContext();
533        for (Object obj : rootContext) {
534            if (obj instanceof SingleFacilityBuilder) {
535                return (SingleFacilityBuilder) obj;
536            }
537        }
538        return null;
539    }

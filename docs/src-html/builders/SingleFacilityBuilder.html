104        @ScheduledMethod(start = 1.0, interval = 1)
105        public void dailyEvents() {
106                if (facility.getPopulationSize() != 0) {
107                        region.doPopulationTally();
108                        region.logDailyPopulationStats();
109                }
110                double dailyPrevalence = 0.0;
111        int count = 0;
112        for (Facility f : region.getFacilities()) {
113            for (FacilityOutbreak outbreak : f.getOutbreaks()) {
114                dailyPrevalence += outbreak.getPrevalence();
115                count++;
116            }
117        }
118        if (count > 0) {
119            dailyPrevalenceSamples.add(dailyPrevalence / count);
120        }
121        
122        int colonized = 0;
123        for (Person p : facility.getCurrentPatients()) {
124                if (p.personDiseases.get(0).isColonized()) {
125                        colonized++;
126                }
127        }
128        // do the same for p.personDiseases.get(0).isDetected()
129       int detected = 0;
130       for (Person p : facility.getCurrentPatients()) {
131           
132           if (p.personDiseases.get(0).isDetected()) {
133           detected++;
134       }
135       }
136       
137       int isolated = 0;
138       for (Person p : facility.getCurrentPatients()) {
139           if (p.isIsolated()) {
140               isolated++;
141       }
142       }
143         
144        
145        dailyPrev.add(facility.getPopulationSize() + "," + colonized  + "," + detected +"," + isolated + ",\n");
146        }

148        public void setupAgents() {
149                // System.out.println("Setting up AGENTS");
150
151                int numDiseases = 1;
152                int[] diseaseList = { 1 }; // Generic disease type ID
153                for (int i = 0; i < numDiseases; i++) {
154                        Disease disease = new Disease();
155                        disease.setSimIndex(i);
156                        disease.setType(diseaseList[i]);
157                        region.getDiseases().add(disease);
158                        disease.setDiseaseName("CRE");
159                }
160
161                int[] facilitySize = { 75 };
162                int[] facilityType = { 0 };
163                double[] meanLOS = { 27.1199026 };
164
165                for (int i = 0; i < region.getFacilities().size(); i++) {
166                        Facility f = region.getFacilities().get(i);
167                        f.setType(facilityType[i]);
168                        f.setAvgPopTarget(facilitySize[i]);
169                        f.setMeanLOS(meanLOS[i]);
170
171                        f.setBetaIsolationReduction(1 - isolationEffectiveness);
172                        f.setNewPatientAdmissionRate(facilitySize[i] / meanLOS[i]);
173
174                        if (doActiveSurveillance) {
175                                f.setTimeBetweenMidstaySurveillanceTests(daysBetweenTests);
176                        }
177
178                        for (Disease d : region.getDiseases()) {
179                                FacilityOutbreak fo = f.addOutbreaks(d);
180                                fo.setDisease(d);
181                                fo.setDiseaseName(d.getDiseaseName());
182                                fo.facility = f;
183                        }
184
185                        for (int j = 0; j < facilitySize[i]; j++) {
186                                region.addInitialFacilityPatient(f);
187                        }
188                        f.admitNewPatient(schedule);
189
190                }
191        }

214        public void doEndBurnInPeriod() {
215
216                region.setInBurnInPeriod(false);
217                region.startDailyPopulationTallyTimer();
218                doActiveSurveillance = doActiveSurveillanceAfterBurnIn;
219                if (doActiveSurveillance) {
220                        for (Facility f : region.getFacilities()) {
221                                f.setTimeBetweenMidstaySurveillanceTests(daysBetweenTests);
222                        }
223                }
224
225                scheduleSimulationEnd();
226        }

257        public void doSimulationEnd() throws IOException {
258            System.out.println("Simulation ending at tick: " + schedule.getTickCount());
259            
260            if (!params.getBoolean("isBatchRun")) {
261                writeDailyPrevToFile();
262                writeDischargedPatientFile();
263            }
264
265
266
267                simulationOutputFile = new PrintWriter("simulation_results.txt");
268                simulationOutputFile.println(
269                                "surveillance_after_burn_in, isolation_effectiveness, days_between_tests, clinical_detections, mean_daily_prevalence, mean_discharge_prevalence, importation_prevalence, number_of_transmissions, sum_daily_infected, sum_daily_clinical_detections"
270                );
271                simulationOutputFile.println(doActiveSurveillanceAfterBurnIn + "," + isolationEffectiveness + ","
272                                + daysBetweenTests + "," + getClinicalDetections() + "," + getMeanDailyPrevalence() + "," + getMeanDischargePrevalence() + "," + getImportationPrevalence() + "," + getNumberOfTransmissions()
273                                + "," + sumDailyInfected + "," + sumDailyClinicalDetections + "\n");
274                simulationOutputFile.flush();
275                simulationOutputFile.close();
276                if (dailyStatsWriter != null) {
277                        dailyStatsWriter.close();
278                }
279                stop = true;
280                System.out.println("Ending simulation at tick: " + schedule.getTickCount());
281
282
283            // writeSimulationResults();
284            region.finishSimulation();
285            // repast.simphony.engine.environment.RunEnvironment.getInstance().endAt(totalTime);
286            repast.simphony.engine.environment.RunEnvironment.getInstance().endRun();
287            System.out.println("Simulation ended.");
288
289        }

301        public int getNumberOfTransmissions() {
302                int numberOfTransmissions = 0;
303                for (Facility f : region.getFacilities()) {
304                        for (FacilityOutbreak outbreak : f.getOutbreaks()) {
305                                numberOfTransmissions += outbreak.getTransmissionsTally();
306                        }
307                }
308                return numberOfTransmissions;
309        }

311        public double getMeanDailyPrevalence() {
312            if (dailyPrevalenceSamples.isEmpty()) {
313                return 0.0;
314            }
315            double sum = 0.0;
316            for (double val : dailyPrevalenceSamples) { sum += val;}
317            return sum / dailyPrevalenceSamples.size();
318        }

320    public double getMeanDischargePrevalence() {
321        double totalDischargePrevalence = 0.0;
322        int count = 0;
323        for (Facility f : region.getFacilities()) {
324            for (FacilityOutbreak outbreak : f.getOutbreaks()) {
325                totalDischargePrevalence += outbreak.getAvgDischargePrevalence();
326                count++;
327            }
328        }
329        return count > 0 ? totalDischargePrevalence / count : 0.0;
330    }

332    public double getImportationPrevalence() {

335        for (Facility f : region.getFacilities()) {

337                totalImportationPrevalence += outbreak.getImportationRate();

339            }

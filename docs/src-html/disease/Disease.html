006public class Disease {
007
008        private int type;
009        private int simIndex;
010        private String diseaseName;

012        
013
014
015        public double getBaselineBetaValue(int facilityType){
016                params = repast.simphony.engine.environment.RunEnvironment.getInstance().getParameters();
017                
018                double longTermAcuteCareBeta = params.getDouble("longTermAcuteCareBeta");
019                double acuteCareBeta = longTermAcuteCareBeta;
020                double nhReduction = (Double) params.getValue("nursingHomeBetaReduction");
021                double nursingHomeBeta = acuteCareBeta / nhReduction;
022
023                double betaVal = 0.0;
024
025                if(facilityType == 0) betaVal = longTermAcuteCareBeta;
026                else if(facilityType == 1) betaVal = acuteCareBeta;
027                else if(facilityType == 2) betaVal = nursingHomeBeta;
028
029                return betaVal;
030        }
031
032        public double getMeanTimeToClinicalDetection(int facilityType){
033                params = repast.simphony.engine.environment.RunEnvironment.getInstance().getParameters();
034                
035                double acuteCareMean = (Double) params.getValue("meanDetectionTime");
036                double longTermAcuteCareMean = (Double) params.getValue("longTermAcuteCareMeanDetectionTime");
037                double nhChangeFactor = (Double) params.getValue("nursingHomeDetectionTimeFactor");
038                double nursingHomeMean = acuteCareMean * nhChangeFactor;
039
040                double t = 0.0;
041
042                if(facilityType == 0) t = longTermAcuteCareMean;
043                else if(facilityType == 1) t = acuteCareMean;
044                else if(facilityType == 2) t = nursingHomeMean;
045
046                return t;
047        }

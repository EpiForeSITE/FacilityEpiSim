001package disease;

003import agents.Person;
004import builders.SingleFacilityBuilder;
005
006import java.io.IOException;
007import java.io.PrintWriter;
008
009import org.apache.commons.math3.distribution.ExponentialDistribution;
010import repast.simphony.engine.schedule.ScheduleParameters;
011import repast.simphony.engine.schedule.ISchedule;
012import repast.simphony.engine.schedule.ISchedulableAction;
013
014public class PersonDisease {
015
016        private Disease disease;
017        private Person person;
018        private boolean colonized = false;
019        private boolean detected = false;
020        private boolean detectedBySurveillance = false; // NEW: track source of detection
021        private double transmissionRateContribution = 1.0;
022        private boolean clinicallyDetectedDuringCurrentStay = false;
023        private boolean initialInfection = false;
024        private int detectionCount = 0;
025
026        private ISchedule schedule;
027        private ExponentialDistribution decolonizationDistribution;
028        private ExponentialDistribution clinicalDetectionDistribution;
029
030        private ISchedulableAction clinicalDetectionAction;
031
032        private static PrintWriter decolWriter;
033        private static PrintWriter clinicalWriter;
034        private static PrintWriter verificationWriter; // NEW: verification log
035        public static int clinicalOutputNum;
036        public static int surveillanceOutputNum; // NEW: count surveillance detections
037
038        static {
039                try {
040                        if (!SingleFacilityBuilder.isBatchRun) {
041                                decolWriter = new PrintWriter("decolonization.txt");
042                                clinicalWriter = new PrintWriter("clinicalDetection.txt");
043                                verificationWriter = new PrintWriter("detection_verification.txt");
044                        }
045                } catch (IOException e) {
046                        e.printStackTrace();
047                }
048        }
049
050        public PersonDisease(Disease disease, Person person, ISchedule schedule) {
051                this.disease = disease;
052                this.person = person;
053                this.schedule = repast.simphony.engine.environment.RunEnvironment.getInstance().getCurrentSchedule();
054
055                initializeEventDistributions();
056
057                if (person.getCurrentFacility() == null) {
058                        System.err.println("Current facility is null for person: " + person);
059                        return;
060                }
061        }
062
063        public void doDecolonization() {
064                double currentTime = schedule.getTickCount();
065                if (this.person.isNoMoreEvents()) {
066                        return;
067                }
068
069                if (colonized) {
070                        colonized = false;
071                        if (!SingleFacilityBuilder.isBatchRun && decolWriter != null) {
072                                decolWriter.printf("Time: %.2f, Decolonized Patient: %d%n", currentTime, person.hashCode());
073                                decolWriter.flush();
074                        }
075                        if (clinicalDetectionAction != null) {
076                                schedule.removeAction(clinicalDetectionAction);
077                                clinicalDetectionAction = null;
078                        }
079                        person.updateAllTransmissionRateContributions();
080                }
081        }
082
083        public void startDecolonizationTimer() {
084                if (decolonizationDistribution == null) {
085                        System.err.println("Decolonization distribution is not initialized.");
086                        return;
087                }
088                double timeToDecolonization = decolonizationDistribution.sample();
089                ScheduleParameters params = ScheduleParameters.createOneTime(schedule.getTickCount() + timeToDecolonization);
090                schedule.schedule(params, this, "doDecolonization");
091        }
092
093        public void doClinicalDetection() {
094                        // If already detected or clinically detected during current stay, do nothing
095                if (detected || clinicallyDetectedDuringCurrentStay || this.person.isNoMoreEvents()) {
096                        return;
097                }
098
099                double currentTime = schedule.getTickCount();
100                detected = true;
101                detectedBySurveillance = false; // mark as clinical
102                clinicallyDetectedDuringCurrentStay = true;
103
104                incrementDetectionCount();
105
106                if (!SingleFacilityBuilder.isBatchRun && clinicalWriter != null) {
107                        clinicalWriter.printf("Time: %.2f, Detected Patient: %d, DetectionCount: %d%n", currentTime,
108                                        person.hashCode(), getDetectionCount());
109                        clinicalWriter.flush();
110                }
111                clinicalOutputNum++;
112
113                // Verification log for detection source
114                if (!SingleFacilityBuilder.isBatchRun && verificationWriter != null) {
115                        verificationWriter.printf("Time: %.2f, Patient: %d, Source: CLINICAL, Colonized: %b, DetectionCount: %d%n",
116                                        currentTime, person.hashCode(), colonized, getDetectionCount());
117                        verificationWriter.flush();
118                }
119
120                if (!person.isIsolated() && disease.isolatePatientWhenDetected()) {
121                        person.isolate();
122                        person.updateAllTransmissionRateContributions();
123                }
124                if (clinicalDetectionAction != null) {
125                        schedule.removeAction(clinicalDetectionAction);
126                        clinicalDetectionAction = null;
127                }
128        }
129
130        /**
131         * Mark this PersonDisease as detected by surveillance testing. Writes to
132         * verification log and performs isolation if configured.
133         */
134        public void setDetectedBySurveillance() {
135                if (detected)
136                        return; // already detected by some source
137                detected = true;
138                detectedBySurveillance = true;
139                surveillanceOutputNum++;
140                double currentTime = schedule.getTickCount();
141                // Verification log
142                if (!SingleFacilityBuilder.isBatchRun && verificationWriter != null) {
143                        verificationWriter.printf(
144                                        "Time: %.2f, Patient: %d, Source: SURVEILLANCE, Colonized: %b, DetectionCount: %d%n", currentTime,
145                                        person.hashCode(), colonized, getDetectionCount());
146                        verificationWriter.flush();
147                }
148                if (!person.isIsolated() && disease.isolatePatientWhenDetected()) {
149                        person.isolate();
150                        person.updateAllTransmissionRateContributions();
151                }
152                if (clinicalDetectionAction != null) {
153                        schedule.removeAction(clinicalDetectionAction);
154                        clinicalDetectionAction = null;
155                }
156        }
157
158        public void startClinicalDetectionTimer() {
159                if (detected || clinicallyDetectedDuringCurrentStay) {
160                        return;
161                }
162                if (clinicalDetectionAction != null) {
163                        schedule.removeAction(clinicalDetectionAction);
164                        clinicalDetectionAction = null;
165                }
166
167                double meanTimeToClinicalDetection = disease
168                                .getMeanTimeToClinicalDetection(person.getCurrentFacility().getType());
169                double timeToDetection = clinicalDetectionDistribution.sample();
170
171                ScheduleParameters params = ScheduleParameters.createOneTime(schedule.getTickCount() + timeToDetection);
172                clinicalDetectionAction = schedule.schedule(params, this, "doClinicalDetection");
173        }
174
175        public void resetClinicalDetectionEvent() {
176                System.out.println("Resetting clinical detection event for person: " + person.hashCode() );
177                detected = false;
178                clinicallyDetectedDuringCurrentStay = false;
179
180                // Clear any pending detection first
181                if (clinicalDetectionAction != null) {
182                        schedule.removeAction(clinicalDetectionAction);
183                        clinicalDetectionAction = null;
184                }
185
186                if (clinicalDetectionDistribution != null) {
187                        double timeToDetection = clinicalDetectionDistribution.sample();
188                        ScheduleParameters params = ScheduleParameters.createOneTime(schedule.getTickCount() + timeToDetection);
189                        clinicalDetectionAction = schedule.schedule(params, this, "doClinicalDetection");
190                }
191        }
192
193        public void colonize() {
194                colonized = true;
195                startDecolonizationTimer();
196                person.updateAllTransmissionRateContributions();
197        }
198
199        public void addAcquisition() {
200                startClinicalDetectionTimer();
201                person.updateAllTransmissionRateContributions();
202        }
203
204        public void incrementDetectionCount() {
205                this.detectionCount++;
206        }
207
208        public int getDetectionCount() {
209                return this.detectionCount;
210        }
211
212        public void updateTransmissionRateContribution() {
213                double score = 1.0;
214                if (person.isIsolated())
215                        score *= person.getCurrentFacility().getBetaIsolationReduction();
216                transmissionRateContribution = score;
217        }
218
219        public void initializeEventDistributions() {
220                if (disease != null && person != null && person.getCurrentFacility() != null) {
221                        double meanTimeToClinicalDetection = disease
222                                        .getMeanTimeToClinicalDetection(person.getCurrentFacility().getType());
223
224                        decolonizationDistribution = new ExponentialDistribution(disease.getAvgDecolonizationTime());
225                        clinicalDetectionDistribution = new ExponentialDistribution(meanTimeToClinicalDetection);
226                } else {
227                        System.err.println("Cannot initialize distributions: disease, person, or current facility is null.");
228                }
229        }
230
231        public boolean isColonized() {
232                return colonized;
233        }

243        public double getTransmissionRateContribution() {
244                return transmissionRateContribution;

264                return detectedBySurveillance;

268                return clinicallyDetectedDuringCurrentStay;
269        }

271        public boolean isInitialInfection() {
272                return initialInfection;
273        }
